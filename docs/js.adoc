=== Funciones

[source,javascript]
.Función básica.
----
function(parametro){
  ...
}
----

[source,javascript]
.Función asignada a una variable.
----
var saludar = function(parametro){
  console.log("Hola "+parametro+"!");
}
----

[source,javascript]
.Named function.
----
function saludar(parametro){
  console.log("Hola "+parametro+"!");
}
----

[source,javascript]
.Clojure/Lambda.
----
var suma = function(x){
  return function(y){
    return x + y;
  }
}

var resultado = suma(1)(2)

console.log(resultado);
----

.Ejemplo de for.
----
var guitarristas = ["hendrix","slash","pass"];

for(var i = 0; i < guitarristas.length ;i++){

}
----

[source,javascript]

----
var guitarristas = ["hendrix","slash","pass"];

  console.log(guitarrista);
}
----
